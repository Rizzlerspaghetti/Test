<script>
function goTab(i) {
  document.querySelectorAll('.tab').forEach((t,j)=>t.classList.toggle('active',j===i));
  document.querySelectorAll('.page').forEach((p,j)=>p.classList.toggle('active',j===i));
  if(i===1) setTimeout(drawAll,50);
}

const svg = document.getElementById('lines');
function edge(id, side) {
  const el = document.getElementById(id);
  const ci = document.querySelector('.canvas-inner').getBoundingClientRect();
  const r  = el.getBoundingClientRect();
  return { x: side==='out' ? r.right - ci.left : r.left - ci.left, y: r.top + r.height/2 - ci.top };
}

function drawConn(a, b, id, cls='') {
  const p1 = edge(a,'out'), p2 = edge(b,'in');
  const mx = (p1.x+p2.x)/2;
  let path = document.getElementById('conn_'+id);
  if(!path) { path = document.createElementNS('http://www.w3.org/2000/svg','path'); path.id='conn_'+id; svg.appendChild(path); }
  path.setAttribute('d',`M${p1.x},${p1.y} C${mx},${p1.y} ${mx},${p2.y} ${p2.x},${p2.y}`);
  path.className = 'conn ' + cls;
}

function drawAll(states={}) {
  drawConn('n1','n2','12',states['12']||'');
  drawConn('n1','n3','13',states['13']||'');
  drawConn('n2','n4','24',states['24']||'');
  drawConn('n3','n4','34',states['34']||'');
  drawConn('n4','n5','45',states['45']||'');
}

function setNode(id, state) {
  document.getElementById('n'+id).className = 'node ' + state;
  const d = document.getElementById('d'+id);
  d.className = 'ndot ' + (state==='running'?'run':state==='done'?'ok':state==='errored'?'err':'');
}

function toast(msg) {
  const t=document.getElementById('toast');
  t.textContent=msg; t.classList.add('show');
  setTimeout(()=>t.classList.remove('show'),2400);
}

async function testKey() {
  const key = document.getElementById('apiKey').value.trim();
  if(!key) { toast('⚠️ Paste API key'); return; }
  try {
    const r = await fetch(`https://generativelanguage.googleapis.com/v1beta/models?key=${key}`);
    toast(r.ok ? '✅ Key works!' : '❌ Invalid key');
  } catch { toast('❌ Error'); }
}

async function executeWorkflow() {
  const key = document.getElementById('apiKey').value.trim();
  const promptText = document.getElementById('prompt').value.trim();
  if(!key || !promptText) { toast('⚠️ Missing info'); return; }

  goTab(2);
  setNode(1,'running');
  // Simple simulation of workflow logic
  setTimeout(()=>setNode(1,'done'), 500);
  
  // Real API Call
  try {
    const res = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-pro:generateContent?key=${key}`, {
      method: 'POST',
      headers: {'Content-Type': 'application/json'},
      body: JSON.stringify({ contents: [{ parts: [{ text: promptText }] }] })
    });
    const data = await res.json();
    const result = data.candidates[0].content.parts[0].text;
    document.getElementById('outputBody').innerHTML = `<div class="out-step"><pre class="code-out">${result}</pre></div>`;
    setNode(3,'done');
  } catch (e) {
    toast('❌ API Error');
  }
}
